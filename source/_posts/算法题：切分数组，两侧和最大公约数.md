### 题面
将一个数组任意排序，并切分成两段，两段和的最大公约数。

### 解法
- 最开始想排序贪心，但不太对
- 实际上因为数据范围1-1000，应该遍历每一种求和情况，DFS
```C++
#include <iostream>
#include <iomanip>
#include <vector>
#include <unordered_set>

using namespace std;

unordered_set<int> mp;
int max_gcd = 1;

int gcd(int x, int y)
{
    if (x < y)
    {
        swap(x, y);
    }
    if (x == 0 || y == 0)
    {
        return -1;
    }
    while (y != 0)
    {
        int temp = x;
        x = y;
        y = (temp % y);
    }
    return x;
}

void dfs(vector<int> &arr, int index, int cur_sum, int sum)
{
    if (index >= arr.size())
    {
        return;
    }
    if (!mp.count(cur_sum))
    {
        mp.insert(cur_sum);
        max_gcd = max(max_gcd, gcd(cur_sum, sum - cur_sum));
    }
    dfs(arr, index + 1, cur_sum + arr[index], sum);
    dfs(arr, index + 1, cur_sum, sum);
}

int main()
{
    vector<int> arr = {1, 1, 2, 1000};
    int sum = 0;
    for (int x : arr)
    {
        sum += x;
    }
    dfs(arr, 0, 0, sum);
    cout << "maxgcd: " << max_gcd;
    return 0;
}
```
